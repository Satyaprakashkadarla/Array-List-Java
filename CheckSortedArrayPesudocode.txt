Function isSorted(arr):
    Input: An array arr of n elements.
    Output: True if the array is sorted in non-decreasing order, False otherwise.

    i := 0
    while i < n - 1 do:
        if arr[i] > arr[i + 1] then:
            return False
        i := i + 1
    end while

    return True
